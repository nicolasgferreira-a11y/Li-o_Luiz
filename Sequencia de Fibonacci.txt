Console.OutputEncoding = System.Text.Encoding.UTF8;
int resultado;
int atual = 1;
int anterior = 0;
int proximo;
Console.WriteLine(" digite uma quantidade de termos:");
int.TryParse(Console.ReadLine()!, out int termos);
for (int i = 0; i <= termos; i++)
{
    proximo = anterior + atual;
    anterior = atual;
    atual = proximo;
    Console.Write($"{anterior}➡️");
}
